$$
Eig(B,\lambda_1)=\langle \begin{pmatrix} 1 \\ 1 \\ 0 \end{pmatrix} \rangle, \ Eig(B,\lambda_2)=\langle \begin{pmatrix} 0 \\ 1 \\ 2 \end{pmatrix} \rangle \text{ und } Eig(B,\lambda_3)=\langle \begin{pmatrix} 0 \\ 2 \\ 1 \end{pmatrix} \rangle
$$
\end{itemize}
